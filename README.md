# Use virtualenv
1. sudo pip install virtualenv
2. virtualenv venv //install venv in the repo
3. source venv/bin/activate
4. exit virtualenv deactivate venv


# Use demo on local
1. cd root-project
2. pip install
3. python manage.py runserver
4. go to http://127.0.0.1:8000/
